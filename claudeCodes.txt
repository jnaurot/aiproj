class DataBuffer:
    """Manages streaming data between nodes"""
    
    def __init__(self, metadata: FileMetadata, buffer_size_mb: int = 100):
        self.metadata = metadata
        self.buffer_size = buffer_size_mb * 1024 * 1024
        self.current_position = 0
        self._buffer: Optional[bytes] = None
    
    async def read_chunk(self, size: int) -> bytes:
        """Read next chunk, refilling buffer if needed"""
        if self._needs_refill():
            await self._refill_buffer()
        
        chunk = self._buffer[self.current_position:self.current_position + size]
        self.current_position += len(chunk)
        return chunk
    
    async def _refill_buffer(self):
        """Strategy depends on file type & access method"""
        match self.metadata.access_method:
            case "local":
                self._buffer = await self._read_local_file()
            case "s3":
                self._buffer = await self._read_s3_chunk()
            case "postgres":
                self._buffer = await self._read_db_batch()

class PipelineExecutor:
    async def execute_pipeline(self, graph: PipelineGraphDTO, run_from: str | None):
        # PHASE 1: Validation (FAIL FAST)
        validation = await self.validator.validate_pre_execution(graph)
        if not validation.valid:
            await self.emit_event(RunFinished(status="failed", errors=validation.errors))
            return
        
        # Emit warnings but continue
        for warning in validation.warnings:
            await self.emit_event(Log(level="warn", message=warning))
        
        # PHASE 2: Execution
        topo_order = self._topological_sort(graph, run_from)
        
        for node_id in topo_order:
            try:
                # Per-node validation BEFORE execution
                await self._validate_node_ready(node_id, context)
                
                # Execute
                result = await self.execute_node(node_id, context)
                
                # Per-node validation AFTER execution
                await self._validate_node_output(node_id, result)
                
                # Update context for downstream nodes
                context.outputs[node_id] = result
                
            except Exception as e:
                await self._handle_node_failure(node_id, e)
                return  # or continue based on error handling policy

class StalenessTracker:
    def calculate_staleness(
        self, 
        node: Node, 
        current_params: dict,
        input_metadata: FileMetadata
    ) -> StalenessResult:
        """Determine if node output is stale"""
        
        # 1. Check if params changed
        stored_hash = self._get_stored_params_hash(node.id)
        current_hash = self._hash_params(current_params)
        
        if stored_hash != current_hash:
            return StalenessResult(
                is_stale=True,
                reason=f"Params changed: {self._diff_params(stored_hash, current_hash)}"
            )
        
        # 2. Check if input changed (recursive check upstream)
        stored_input_hash = self._get_stored_input_hash(node.id)
        current_input_hash = input_metadata.hash
        
        if stored_input_hash != current_input_hash:
            return StalenessResult(
                is_stale=True,
                reason=f"Input data changed (upstream node re-executed)"
            )
        
        # 3. Check if node code/logic changed (version tracking)
        if self._node_version_changed(node):
            return StalenessResult(
                is_stale=True,
                reason="Node implementation updated"
            )
        
        return StalenessResult(is_stale=False)
    
    def propagate_staleness(self, node_id: str, graph: PipelineGraphDTO):
        """Mark all downstream nodes as stale"""
        downstream = self._get_downstream_nodes(node_id, graph)
        
        for downstream_id in downstream:
            self._mark_stale(downstream_id, reason=f"Upstream node {node_id} changed")

class ResourceMonitor:
    def __init__(self, max_memory_mb: int = 4096):
        self.max_memory = max_memory_mb
        self.current_usage = 0
    
    async def execute_with_limit(self, node_id: str, executor: Callable):
        estimated = self._estimate_node_memory(node_id)
        
        if self.current_usage + estimated > self.max_memory:
            # Wait for memory or fail
            await self._wait_for_memory(estimated)
        
        try:
            self.current_usage += estimated
            result = await executor()
            return result
        finally:
            self.current_usage -= estimated
            self._cleanup_buffers()

async def execute_pipeline_parallel(self, graph: PipelineGraphDTO):
    """Execute independent nodes in parallel"""
    
    topo_levels = self._get_execution_levels(graph)  # [[level0_nodes], [level1_nodes], ...]
    
    for level in topo_levels:
        # All nodes in a level can execute in parallel
        tasks = [self.execute_node(node_id, context) for node_id in level]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # Check for failures
        for node_id, result in zip(level, results):
            if isinstance(result, Exception):
                await self._handle_failure(node_id, result)
                return
